%% https://ctan.math.illinois.edu/macros/latex/required/graphics/graphics.pdf
%% SEE: \graphicspath pag. 11
\makeatletter%
\def\SetInputPaths#1{%
	\def\MyPaths@input{#1}%
	\ifx\input@path\@defined%
		\let\input@path\MyPaths@input%
		\typeout{@@@@@@@@@@@@@ Input path variable defined, setting paths ...}%
	\else%
		\typeout{@@@@@@@@@@@@@ Input path variable not defined, creating and setting paths ...}%
		\gdef\input@path{\MyPaths@input}%
	\fi%
}%
\makeatother%

\newcommand{\setprintabletext}[1]{
	
	\ignorespaces\lowercase{\def\tmp{#1}}\unskip
	
	\ifx\tmp\trueValue
		\global\def\printabletext{true}
	\else
		\global\def\printabletext{false}
	\fi
}

\newcommand{\setprintableimages}[1]{
	
	\ignorespaces\lowercase{\def\tmp{#1}}\unskip
	
	\ifx\tmp\trueValue
		\global\def\printableimages{true}
	\else
		\global\def\printableimages{false}
	\fi
}

\newcommand{\checkforceprintable}{
	
	\ifdefined\forceprint
		\setprintabletext{true}
		\setprintableimages{true} 
	\else
		\setprintabletext{false}
		\setprintableimages{false} 
	\fi
}

\newcommand{\twosidelayout}[1][]{
	
	\ignorespaces\lowercase{\def\tmp{#1}}\unskip

	\ifx\tmp\empty
		\twosidetrue
	\else
	
		\ifx\tmp\falseValue

			\twosidetrue
		\fi
	\fi
}
